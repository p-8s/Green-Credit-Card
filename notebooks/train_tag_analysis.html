check_woe <- function(var) {
  tab.var <- table(as.factor(var), good)
  df.var <- as.data.frame.matrix(tab.var)
  df.prop.var <- as.data.frame.matrix(prop.table(tab.var, 1))
  df.var$woe <- log(df.prop.var$"1" / df.prop.var$"0")
  return(df.var)
